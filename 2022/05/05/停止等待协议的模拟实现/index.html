еҚҸи®®еҺҹзҗҶ

и®ҫи®ЎжҖқи·Ҝ

еҹәжң¬дј иҫ“

жң¬ж¬Ўе®һйӘҢжҲ‘йҮҮз”ЁдәҶзЁӢеәҸжЁЎжӢҹзҡ„ж–№ејҸе®һзҺ°гҖӮеҸ‘йҖҒж–№е’ҢжҺҘж”¶ж–№йғҪдёәдёҖдёӘж•°з»„пјҢдј иҫ“иҝҮзЁӢеҚідёәеҸ‘йҖҒж–№ж•°з»„еҗ‘жҺҘж”¶ж–№ж•°з»„дј йҖ’ж•°жҚ®пјҢе№¶дҪҝз”ЁйҡҸжңәж•°з”ҹжҲҗзҡ„ж–№ејҸжЁЎжӢҹдј иҫ“иҝҮзЁӢдёӯеҸҜиғҪеҮәзҺ°зҡ„е·®й”ҷпјҢе№¶дё”дј иҫ“ж—¶й—ҙд№ҹдёәдёҖдёӘеңЁ 50 - 150 ms й—ҙзҡ„йҡҸжңәж•°пјҢз”ЁдёҖдёӘиҝӯд»ЈеҷЁжқҘжЁЎжӢҹжӯӨж—¶еҸ‘йҖҒж–№зҡ„ж•°жҚ®дҪҚзҪ®гҖӮ

дј иҫ“иҝҮзЁӢдёәдёҖдёӘеҫӘзҺҜиҜӯеҸҘпјҢз»ҲжӯўжқЎд»¶дёәеҸ‘йҖҒиҝӯд»ЈеҷЁеҲ°иҫҫеҸ‘йҖҒж–№ж•°з»„зҡ„жң«е°ҫдҪҚзҪ®гҖӮ

йҖүжӢ©йҮҚдј 

жҜҸж¬ЎиҝӣиЎҢдј иҫ“пјҢдј иҫ“ж—¶й—ҙе°ұдјҡзҙҜеҠ пјҢеңЁдёҖиҪ®дј иҫ“зҡ„жңҖеҗҺеҲӨж–ӯзҙҜи®Ўж—¶й—ҙжҳҜеҗҰи¶…иҝҮдәҶи§„е®ҡ toutпјҢеҰӮжһңжңӘи¶…иҝҮпјҢд»ЈиЎЁж•°жҚ®жҲҗеҠҹжҺҘж”¶пјҢеҸ‘йҖҒиҝӯд»ЈеҷЁиҮӘеўһ 1гҖӮ еҗҰеҲҷпјҢдёҚиҝӣиЎҢж“ҚдҪңпјҢдёӢдёҖж¬ЎеҫӘзҺҜе°ҶдјҡеҶҚж¬Ўе°қиҜ•еҸ‘йҖҒиҜҘж•°жҚ®гҖӮ

иҲҚејғйҮҚеӨҚеё§

еӯҳеңЁиҝҷд№ҲдёҖз§Қжғ…еҶөпјҢжҺҘж”¶ж–№жҲҗеҠҹжҺҘж”¶дәҶеҸ‘йҖҒж–№зҡ„ж•°жҚ®пјҢе№¶иҝ”еӣһдәҶдёҖдёӘ ACK её§пјҢдҪҶжҳҜжӯӨ ACK её§иҝҳжңӘеҲ°иҫҫеҸ‘йҖҒж–№е°ұе·Іиў«е…¶еҲӨе®ҡдёәи¶…ж—¶пјҢйӮЈд№ҲеҸ‘йҖҒж–№е°ҶдјҡйҮҚж–°еҸ‘йҖҒдёҠдёҖж¬Ўзҡ„ж•°жҚ®её§пјҢиӢҘиҜҘж•°жҚ®её§жҲҗеҠҹеҲ°иҫҫжҺҘж”¶ж–№пјҢйӮЈд№ҲжҺҘж”¶ж–№йңҖиҰҒе°ҶиҜҘж•°жҚ®её§иҲҚејғпјҲеӣ жӯӨдёҠдёҖж¬Ўдј иҫ“ж—¶жҺҘж”¶ж–№е·ІжҲҗеҠҹжҺҘеҸ—дәҶиҜҘж•°жҚ®её§пјүгҖӮе®һзҺ°ж–№жі•жҳҜеҲӨж–ӯжҺҘеҸ—ж•°з»„зҡ„жңҖеҗҺдёҖдёӘе…ғзҙ пјҲеҚідёҠдёҖж¬ЎжҺҘж”¶зҡ„е…ғзҙ пјүжҳҜеҗҰдёҺжӯӨж—¶жҺҘеҸ—зҡ„зӣёеҗҢпјҢиӢҘзӣёеҗҢпјҢеҲҷдёҚжҺҘж”¶пјҢе№¶йҮҚж–°еҸ‘йҖҒ ACK её§гҖӮ

д»Јз Ғе®һзҺ°

#include<iostream>
#include<string>
#include<vector>
#include<Windows.h>
using namespace std;

struct myData
{
	int bin;
	string str;
	myData(int b, string s) : bin(b), str(s) {}
};

class ARQ {
public:
	void transmitData(vector<myData>& reciver, vector<myData>& sender) {
		srand((unsigned int)time(NULL));
		auto itSender = sender.cbegin();
		while (itSender != sender.cend()) {
			int sendBin = rand() % 2; // дј иҫ“иҝҮзЁӢж•°жҚ®еҸҜиғҪеҮәй”ҷ
			double transTime = rand() % 101 + 50; // йҡҸжңәдј иҫ“ж—¶й—ҙдёә 50 - 150 ms
			myData sendData(sendBin, itSender->str); // еҸ–ж•°жҚ®её§
			cout << "sender: жӯЈеңЁеҸ‘йҖҒз¬¬ " << itSender - sender.cbegin() << " её§..." << endl;
			Sleep(transTime); // ж•°жҚ®дј иҫ“иҝҮзЁӢ
			bool isRight = sendData.bin; // жЈҖй”ҷ
			if (!reciver.empty() && reciver.back().str == sendData.str) { // иҲҚејғйҮҚеӨҚеё§
				cout << "reciver: дёўејғйҮҚеӨҚеё§пјҢеҮҶеӨҮйҮҚж–°еҸ‘йҖҒ ACK её§гҖӮ" << endl;
				isRight = true;
				transTime += rand() % 101 + 50;
				cout << "reciver: жӯЈеңЁеҸ‘йҖҒ ACK её§..." << endl;
				Sleep(transTime);
			}
			else if (isRight) { // ж•°жҚ®жӯЈзЎ®
				reciver.emplace_back(sendData); // жҺҘж”¶ж•°жҚ®
				cout << "reciver: ж•°жҚ®жҲҗеҠҹжҺҘж”¶пјҢеҮҶеӨҮеҸ‘йҖҒ ACK её§гҖӮ" << endl;
				transTime += rand() % 101 + 50;
				cout << "reciver: жӯЈеңЁеҸ‘йҖҒ ACK её§..." << endl;
				Sleep(transTime);
			}
			else { // ж•°жҚ®й”ҷиҜҜ
				cout << "reciver: ж•°жҚ®й”ҷиҜҜпјҢеҸ‘йҖҒ NAK её§гҖӮ" << endl;
				transTime += rand() % 101 + 50;
				cout << "reciver: жӯЈеңЁеҸ‘йҖҒ NAK её§..." << endl;
				Sleep(transTime);
			}

			if (transTime < 250) { // жңӘи¶…ж—¶
				if (isRight) {
					cout << "sender: жҲҗеҠҹжҺҘж”¶ ACK её§пјҢеҮҶеӨҮеҸ‘йҖҒдёӢдёҖеё§гҖӮ" << endl << endl;
					++itSender;
				}
				else {
					cout << "sender: жҲҗеҠҹжҺҘж”¶ NAK её§пјҢеҮҶеӨҮйҮҚдј гҖӮ" << endl << endl;
				}
			}
			else { // дј иҫ“и¶…ж—¶
				cout << "sender: и¶…ж—¶пјҢеҮҶеӨҮйҮҚдј гҖӮ" << endl << endl;
			}
		}
	}
};

int main() {
	ARQ arq;
	vector<myData> reciver; // еҲқе§Ӣж—¶жҺҘж”¶ж–№дёәз©ә
	vector<myData> sender = { myData(1, "Hello"),myData(1,"World"), myData(1, "!") }; // еҲқе§ӢеҢ–еҸ‘йҖҒж–№
	arq.transmitData(reciver, sender);
	cout << "---------------------------------" << endl;
	cout << "ж•°жҚ®дј иҫ“з»“жқҹпјҢд»ҘдёӢдёәжҺҘж”¶иҖ…ж”¶еҲ°зҡ„ж•°жҚ®пјҡ" << endl;
	for (const auto& s : sender) {
		cout << s.bin << " " << s.str << endl;
	}
}
