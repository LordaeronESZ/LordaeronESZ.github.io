з”ЁDFSи§ЈеҶіжңҖз»Ҳе№»жғі13-2ж—¶й’ҹи°ңйўҳ

еҸ‘иЎЁдәҺ2022-08-08|жӣҙж–°дәҺ2024-07-01|ж•°жҚ®з»“жһ„дёҺз®—жі•
|еӯ—ж•°жҖ»и®Ў:808|йҳ…иҜ»ж—¶й•ҝ:3еҲҶй’ҹ|йҳ…иҜ»йҮҸ:

жңҖиҝ‘еңЁиЎҘ XGP дёӯзҡ„жңҖз»Ҳе№»жғі13-2ж—¶пјҢйҒҮеҲ°дёҖдёӘж—¶й’ҹи°ңйўҳпјҢж„ҹи§үжҢәжңүж„ҸжҖқпјҢе°ұеғҸе°қиҜ•з”Ёжҗңзҙўз®—жі•е°Ҷе…¶и§ЈеҶігҖӮ

й—®йўҳжҸҸиҝ°

еҰӮдёӢеӣҫжүҖзӨәпјҢжңүдёҖдёӘж—¶й’ҹпјҢеҢ…еҗ«дёӘз»“зӮ№пјҢжҜҸдёӘиҠӮзӮ№жңүдёҖдёӘж•°еӯ—ж ҮиҜҶпјҢзҺ©е®¶жңҖејҖе§ӢеҸҜд»Ҙд»»ж„ҸйҖүжӢ©дёҖдёӘз»“зӮ№пјҢйҖүжӢ©еҗҺпјҢиҜҘз»“зӮ№иў«ж¶ҲйҷӨдё”жҢҮй’ҲдјҡжҢҮеҗ‘иҜҘз»“зӮ№зҡ„дҪҚзҪ®пјҢж №жҚ®иҜҘиҠӮзӮ№зҡ„ж•°еӯ—еҖј n еҲҶиЈӮдёәдёӨж №жҢҮй’ҲеҲҶеҲ«еҗ‘йЎәж—¶й’Ҳж–№еҗ‘е’ҢйҖҶж—¶й’Ҳж–№еҗ‘ж—ӢиҪ¬ n дёӘзҡ„еҚ•дҪҚй•ҝеәҰгҖӮжӯӨеҗҺжҜҸж¬ЎзҺ©е®¶еҸӘиғҪйҖүжӢ©жҢҮй’ҲжҢҮеҗ‘зҡ„з»“зӮ№пјҢйҖүжӢ©з»“зӮ№еҗҺз»“зӮ№иў«ж¶ҲйҷӨпјҢдёӨжҢҮй’ҲеҗҲе№¶жҢҮеҗ‘йҖүжӢ©з»“зӮ№зҡ„дҪҚзҪ®е№¶жҢүдёҠиҝ°жҸҸиҝ°иҝӣиЎҢеҲҶиЈӮе’Ңж—ӢиҪ¬пјҢзҺ©е®¶йңҖиҰҒе°ҶжүҖжңүиҠӮзӮ№ж¶ҲйҷӨжүҚиғҪиғңеҲ©гҖӮ

жіЁпјҡзҺ©е®¶ж— жі•йҖүжӢ©е·Із»Ҹиў«ж¶ҲйҷӨзҡ„з»“зӮ№пјҢиӢҘеҲҶиЈӮж—ӢиҪ¬еҗҺзҡ„дёӨжҢҮй’ҲеқҮдҪҚдәҺе·Іиў«ж¶ҲйҷӨзҡ„з»“зӮ№дҪҚзҪ®пјҢеҲҷеҲӨе®ҡжёёжҲҸеӨұиҙҘгҖӮ

з®—жі•жҖқи·Ҝ

жң¬й—®йўҳеҫҲе®№жҳ“жғіеҲ°еҲ©з”Ёж·ұеәҰдјҳе…ҲжҗңзҙўжқҘи§ЈеҶіпјҢйҖүжӢ©дёҖдёӘз»“зӮ№дҪңдёәејҖе§ӢпјҢеҰӮз¬¬дёҖж¬ЎйҖүжӢ© 12 зӮ№й’ҹдҪҚзҪ®зҡ„з»“зӮ№пјҢпјҲд»ҘдёӢдёәдәҶж–№дҫҝпјҢжҢүз»“зӮ№еңЁж—¶й’ҹдёӯжҺ’еёғдҪҚзҪ® n з§°дҪңз»“зӮ№ nпјүиҜҘз»“зӮ№еҖјдёә 5пјҢеҲҷйҖүдёӯеҗҺеҲҶеҲ«еҗ‘йЎәж—¶й’Ҳе’ҢйҖҶж—¶й’Ҳж–№еҗ‘ж—ӢиҪ¬еҲ°иҫҫз»“зӮ№ 5 е’Ң з»“зӮ№ 7пјҢиҝҷе°ұдә§з”ҹдәҶдёӨдёӘеҲҶж”ҜпјҲзӣёеҪ“дәҺдәҢеҸүж ‘зҡ„е·ҰеҸіеӯҗж ‘пјүпјҢеҲҶеҲ«йҖүжӢ©иҝҷдёӨдёӘз»“зӮ№з»§з»ӯжҗңзҙўпјҢиӢҘз»“зӮ№еҲ°иҫҫдәҶдёҖдёӘе·Іиў«и®ҝй—®иҝҮзҡ„з»“зӮ№пјҲеҚіиҜҘз»“зӮ№е·Іиў«ж¶ҲйҷӨпјүпјҢеҲҷз»ҲжӯўиҜҘж–№еҗ‘дёҠзҡ„жҗңзҙўпјҢе№¶иҝӣиЎҢеӣһжәҜпјҢе°Ҷи·Ҝеҫ„дёҠзҡ„иҜҘз»“зӮ№еҲ йҷӨпјҢе№¶е°Ҷи®ҝй—®ж Үеҝ—еӨҚеҺҹгҖӮ

иӢҘи·Ҝеҫ„дёҠзҡ„з»“зӮ№дёӘж•°е·Із»ҸиҫҫеҲ° 12пјҢеҚіжүҖжңүиҠӮзӮ№еқҮиў«жҲҗеҠҹж¶ҲйҷӨпјҢеҲҷиҜҘи·Ҝеҫ„дёәдёҖдёӘи§Ји·Ҝеҫ„пјҢе°ҶиҜҘз»“жһңдҝқеӯҳе№¶еӣһжәҜз»§з»ӯиҝӣиЎҢжҗңзҙўпјҢзӣҙеҲ°е°қиҜ•дәҶжүҖжңүеҸҜиғҪжҖ§пјҢз®—жі•з»“жқҹгҖӮ

е®Ңж•ҙд»Јз Ғ

#include<iostream>
#include<vector>
using namespace std;

class Solution {
private:
    int n;
    vector<int> path; // жҗңзҙўи·Ҝеҫ„
    vector<vector<int>> res; // жҲҗеҠҹи·Ҝеҫ„
    void dfs(vector<int>& sequences, vector<bool>& visited, int ind) {
        if (visited[ind]) { // е·Іи®ҝй—®иҝҮ
            return;
        }
        path.emplace_back(ind);
        visited[ind] = true;

        if (path.size() == n) { // жҲҗеҠҹи®ҝй—®жүҖжңүе…ғзҙ 
            res.emplace_back(path);
        }

        int p1 = (ind - sequences[ind] + n) % n; // йҖҶж—¶й’Ҳж–№еҗ‘ж—ӢиҪ¬еҗҺзҡ„дёӢж Ү
        int p2 = (ind + sequences[ind]) % n; // йЎәж—¶й’Ҳж–№еҗ‘ж—ӢиҪ¬еҗҺзҡ„дёӢж Ү

        if (p1 == p2) {
            dfs(sequences, visited, p1);
        }
        else {
            dfs(sequences, visited, p1);
            dfs(sequences, visited, p2);
        }

        // еӣһжәҜ
        path.pop_back();
        visited[ind] = false;
    }
public:
    vector<vector<int>> clockPuzzle(vector<int>& sequences) {
        n = sequences.size();
        vector<bool> visited;
        for (int i = 0; i < n; ++i) { // йҖүеҸ–дёҖдёӘз»“зӮ№дҪңдёәејҖе§Ӣз»“зӮ№
            visited.assign(n, false); // йҮҚзҪ®visitedж•°з»„ 
            dfs(sequences, visited, i);
        }
        return res;
    }
};

int main() {
    Solution S;
    vector<int> sequences = { 5,5,3,6,2,1,4,3,4,5,3,4 };
    auto res = S.clockPuzzle(sequences);
    for (const auto& re : res) {
        for (const auto& r : re) {
            cout << r << " ";
        }
        cout << endl << endl;
    }
}
